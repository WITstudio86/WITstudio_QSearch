
# 累加等累计算

## 通过 fold 方法累加

```rust
use num::PrimInt;


// 累乘
fn get_product<T:PrimInt>(vec: &Vec<T>) -> T {
    vec.iter().fold(T::one(), |acc , n| acc * *n)
}

// 累加
fn get_sum<T:PrimInt>(vec: &Vec<T>) -> T {
    vec.iter().fold(T::zero(), |acc , n| acc + *n)
}

fn main() {
    let vec = vec![1,2,3,4,4];
    // 累乘结果
    let result = get_product(&vec);// 96
    println!("result: {}", result);

    let result = get_sum(&vec);// 14
    println!("result: {}", result);
}
```

```rust
use std::collections::HashMap;
use std::io::stdin;

fn main(){
    let mut score:HashMap<String, u32> = HashMap::new();

    // 插入键值对
    score.insert(String::from("张三"), 100);
    score.insert(String::from("李四"), 50);
    //修改李四
    score.insert(String::from("李四"), 90);
    // 删除张三
    score.remove(&String::from("张三"));
    println!("{:?}", score);

    let mut input = String::new();
    stdin().read_line(&mut input).unwrap();
    let input:String = input.trim().parse().unwrap();
    
    // 输出 input 所对应的字段
    println!("{}", score.get(&input).unwrap());
}
```

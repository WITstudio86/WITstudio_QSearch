
## 限制整数

```rust
use num::PrimInt;
use std::convert::From;

#[derive(Debug)]
struct Number(i32);

impl <T:PrimInt> From<T> for Number {
    fn from(val : T) -> Self{
        Number(val.to_i32().unwrap())
    }
}

fn main() {
    let number = Number::from(100_usize);
    println!("{:?}", number);
    let number: Number = 100_usize.into();
    println!("{:?}", number);
}
```

> `num = "0.4.3"`

## From 和 Into

```rust
use num::PrimInt;
use std::convert::From;

#[derive(Debug)]
struct Number(i32);

impl <T:PrimInt> From<T> for Number {
    fn from(val : T) -> Self{
        Number(val.to_i32().unwrap())
    }
}

fn main() {
    let number = Number::from(100_usize);
    println!("{:?}", number);
    let number: Number = 100_usize.into();
    println!("{:?}", number);
}
```

## try_from 和 try_into

用在容易出错的地方 , 返回的是 Result

```rust
use std::convert::{TryFrom,TryInto};
#[derive(Debug)]
struct EvenNumber(i32);

impl TryFrom<i32> for EvenNumber {
    type Error = ();
    fn try_from(value: i32) -> Result<Self, Self::Error> {
        if value % 2 == 0 {
            Ok(EvenNumber(value))   
        }else{
            Err(())
        }
    }
}

fn main() {
    let result: Result<EvenNumber, ()> = 8i32.try_into();
    println!("{:?}", result);
    let result = EvenNumber::try_from(5i32);
    println!("{:?}", result);
}
```

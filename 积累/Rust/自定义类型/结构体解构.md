
```rust
fn rect_area(rect : Rectangle) -> f32{
    // 嵌套解构
    let Rectangle {
        top_left: Point { x: left_x, y: top_y },
        bottom_right: Point { x: right_x, y: bottom_y },
    } = rect;
    let width = right_x - left_x;
    let height = top_y - bottom_y;
    // 计算面积
    width * height
}

fn square(point: Point , length : f32) -> Rectangle {
    let Point{x : left_x , y: top_y} = point;
    Rectangle {
        top_left: Point { x: left_x , y: top_y },
        bottom_right: Point { x: left_x + length , y: top_y - length },
    }
}
```

> 注意结构体解构的时候是需要写字段名字结构给谁的 , 且要有结构体名称
